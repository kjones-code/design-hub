let links=[],atConfig=null,isLoading=false;

function initials(n){return(n||'?').split(' ').map(p=>p[0]||'').join('').slice(0,2).toUpperCase();}


function openForm(){

  document.getElementById('form-title').textContent='Add a new link';

  document.getElementById('form-submit-btn').textContent='Save link';

  document.getElementById('f-id').value='';

  ['f-title','f-url','f-designer','f-desc','f-launch'].forEach(id=>document.getElementById(id).value='');

  document.getElementById('f-status').selectedIndex=0;

  document.getElementById('f-category').selectedIndex=0;

  document.getElementById('form-wrap').classList.add('open');

}


function openEditForm(id){

  const l=links.find(x=>x.id===id);

  if(!l)return;

  document.getElementById('form-title').textContent='Edit link';

  document.getElementById('form-submit-btn').textContent='Update link';

  document.getElementById('f-id').value=l.id;

  document.getElementById('f-title').value=l.title;

  document.getElementById('f-url').value=l.url;

  document.getElementById('f-designer').value=l.designer;

  document.getElementById('f-desc').value=l.desc||'';

  document.getElementById('f-launch').value=l.launch||'';

  document.getElementById('f-status').value=l.status||'Happening Now';

  document.getElementById('f-category').value=l.category;

  document.getElementById('form-wrap').classList.add('open');

}


function closeForm(){document.getElementById('form-wrap').classList.remove('open');}

function showConfig(){document.getElementById('config-modal').classList.add('open');}

function hideConfig(){document.getElementById('config-modal').classList.remove('open');}


function saveConfig(){

  const token=document.getElementById('at-token').value.trim();

  const base=document.getElementById('at-base').value.trim();

  const table=document.getElementById('at-table').value.trim()||'Design Links';

  if(!token||!base){alert('Please enter your token and base ID.');return;}

  atConfig={token,base,table};

  localStorage.setItem('pcdh-at-config',JSON.stringify(atConfig));

  hideConfig();

  document.getElementById('config-banner').style.display='none';

  fetchFromAirtable();

}


async function fetchFromAirtable(){

  if(!atConfig)return;

  isLoading=true;

  document.getElementById('loading').style.display='block';

  document.getElementById('board').style.display='none';

  try{

    const r=await fetch(`https://api.airtable.com/v0/${atConfig.base}/${encodeURIComponent(atConfig.table)}`,{headers:{Authorization:`Bearer ${atConfig.token}`}});

    const data=await r.json();

    if(data.records){

      links=data.records.map(rec=>({

        id:rec.id,

        title:rec.fields['Project name']||'',

        status:rec.fields['Status']||'Happening Now',

        url:rec.fields['URL']||'',

        designer:rec.fields['Designer']||'',

        category:rec.fields['Category']||'Route Optimization',

        desc:rec.fields['Description']||'',

        launch:rec.fields['Launch Date']||'',

      }));

    }

  }catch(e){console.error('Airtable fetch error:',e);}

  isLoading=false;

  document.getElementById('loading').style.display='none';

  document.getElementById('board').style.display='grid';

  render();

}


async function addToAirtable(link){

  if(!atConfig)return null;

  const r=await fetch(`https://api.airtable.com/v0/${atConfig.base}/${encodeURIComponent(atConfig.table)}`,{

    method:'POST',headers:{Authorization:`Bearer ${atConfig.token}`,'Content-Type':'application/json'},

    body:JSON.stringify({fields:{'Project name':link.title,'Status':link.status,'URL':link.url,'Designer':link.designer,'Category':link.category,'Description':link.desc,'Launch Date':link.launch}})

  });

  const data=await r.json();return data.id||null;

}


async function updateInAirtable(link){

  if(!atConfig)return;

  await fetch(`https://api.airtable.com/v0/${atConfig.base}/${encodeURIComponent(atConfig.table)}/${link.id}`,{

    method:'PATCH',headers:{Authorization:`Bearer ${atConfig.token}`,'Content-Type':'application/json'},

    body:JSON.stringify({fields:{'Project name':link.title,'Status':link.status,'URL':link.url,'Designer':link.designer,'Category':link.category,'Description':link.desc,'Launch Date':link.launch}})

  });

}


async function deleteFromAirtable(id){

  if(!atConfig)return;

  await fetch(`https://api.airtable.com/v0/${atConfig.base}/${encodeURIComponent(atConfig.table)}/${id}`,{method:'DELETE',headers:{Authorization:`Bearer ${atConfig.token}`}});

}


async function submitForm(){

  const title=document.getElementById('f-title').value.trim();

  const url=document.getElementById('f-url').value.trim();

  const designer=document.getElementById('f-designer').value.trim();

  if(!title||!url||!designer){alert('Please fill in project name, URL, and designer.');return;}

  const editId=document.getElementById('f-id').value;

  const linkData={title,status:document.getElementById('f-status').value,url,designer,category:document.getElementById('f-category').value,desc:document.getElementById('f-desc').value.trim(),launch:document.getElementById('f-launch').value.trim()};

  if(editId){

    const idx=links.findIndex(l=>l.id===editId);

    if(idx>-1){

      links[idx]={...links[idx],...linkData};

      if(atConfig)await updateInAirtable(links[idx]);

      if(!atConfig)localStorage.setItem('pcdh-local-links',JSON.stringify(links));

    }

  }else{

    const link={id:'local-'+Date.now(),...linkData};

    if(atConfig){const atId=await addToAirtable(link);if(atId)link.id=atId;}

    links.unshift(link);

    if(!atConfig)localStorage.setItem('pcdh-local-links',JSON.stringify(links));

  }

  closeForm();render();

}


async function deleteLink(id){

  if(atConfig)await deleteFromAirtable(id);

  links=links.filter(l=>l.id!==id);

  if(!atConfig)localStorage.setItem('pcdh-local-links',JSON.stringify(links));

  render();

}

    const ma=!fa||l.category===fa;

  });

}
